548 friend bool operator==(const vector &lhs, const vector &rhs) {
549 if (lhs.size() != rhs.size()) {
550 return false;
551 }
552 auto lhs_i = lhs.begin(), rhs_i = rhs.begin();
553 while (lhs_i != lhs.end() && rhs_i != rhs.end()) {
554 if (*lhs_i != *rhs_i) {
555 return false;
556 }
557 lhs_i++, rhs_i++;
558 }
559 return true;

699 template <class ForwardIt>
700 void m_range_insert(iterator pos, ForwardIt first, ForwardIt last, tstl::forward_iterator_tag) {
701 if (first == last)
702 return;
703 const size_type n = tstl::distance(first, last);
704 if (size_type(m_end_of_storage - m_finish) >= n) {
705 const size_type elems_after = end() - pos;
706 pointer old_finish = m_finish;
707 if (elems_after > n) {
708 tstl::_uninitialized_move_a(m_finish - n, m_finish, m_finish, m_alloc);
709 m_finish += n;
710 tstl::move_backward(pos.base(), old_finish - n, old_finish);
711 tstl::copy(first, last, pos);
712 } else {
713 ForwardIt mid = first;
714 tstl::advance(mid, elems_after);
715 tstl::_uninitialized_copy_a(mid, last, m_finish, m_alloc);
716 m_finish += n - elems_after;
717 tstl::_uninitialized_move_a(pos.base(), old_finish, m_finish, m_alloc);
718 m_finish += elems_after;
719 tstl::copy(first, mid, pos);
720 }
721 } else {
722 const size_type len = m_check_len(n);
723 pointer new_start = m_allocate(len);
724 pointer new_finish = new_start;
725 try {
726 new_finish = tstl::_uninitialized_copy_a(m_start, pos.base(), new_start, m_alloc);
727 new_finish = tstl::_uninitialized_copy_a(first, last, new_finish, m_alloc);
728 new_finish = tstl::_uninitialized_copy_a(pos.base(), m_finish, new_finish, m_alloc);
729 } catch (...) {
730 m_destroy(new_start, new_finish);
731 m_deallocate(new_start, len);
732 throw;
733 }
734 m_destroy(m_start, m_finish);
735 m_deallocate(m_start, m_end_of_storage - m_start);
736 m_start = new_start;
737 m_finish = new_finish;
738 m_end_of_storage = new_start + len;
739 }
740 }

747 void m_fill_insert(iterator pos, size_type n, const T &value) {
748 if (n == 0)
749 return;
750 if (size_type(m_end_of_storage - m_finish) >= n) {
751 const size_type elems_after = end() - pos;
752 pointer old_finish = m_finish;
753 if (elems_after > n) {
754 tstl::_uninitialized_move_a(m_finish - n, m_finish, m_finish, m_alloc);
755 m_finish += n;
756 tstl::move_backward(pos.base(), old_finish - n, old_finish);
757 tstl::fill(pos.base(), pos.base() + n, value);
758 } else {
759 m_finish = tstl::_uninitialized_fill_n_a(m_finish, n - elems_after, value, m_alloc);
760 _uninitialized_move_a(pos.base(), old_finish, m_finish, m_alloc);
761 m_finish += elems_after;
762 tstl::fill(pos.base(), old_finish, value);
763 }
764 } else {
765 const size_type len = m_check_len(n);
766 const size_type elems_before = pos - begin();
767 pointer new_start = m_allocate(len);
768 pointer new_finish = new_start;
769 try {
770 tstl::_uninitialized_fill_n_a(new_start + elems_before, n, value, m_alloc);
771 new_finish = nullptr;
772 new_finish = tstl::_uninitialized_copy_a(m_start, pos.base(), new_start, m_alloc);
773 new_finish += n;
774
775 new_finish = tstl::_uninitialized_copy_a(pos.base(), m_finish, new_finish, m_alloc);
776
777 } catch (...) {
778 if (new_finish != nullptr) {
779 m_destroy(new_start + elems_before, new_start + elems_before + n);
780 } else {
781 m_destroy(new_start + elems_before + n);
782 }
783 m_deallocate(new_start, len);
784 throw;
785 }
786 m_destroy(m_start, m_finish);
787 m_deallocate(m_start, m_end_of_storage - m_start);
788 m_start = new_start;
789 m_finish = new_finish;
790 m_end_of_storage = new_start + len;
791 }
792 }

804 template <class... Args>
805 void m_realloc_insert(iterator pos, Args &&...args) {
806 const size_type len = m_check_len(1);
807 pointer old_start = m_start;
808 pointer old_finish = m_finish;
809 const size_type elems_before = pos - begin();
810 pointer new_start = m_allocate(len);
811 pointer new_finish = new_start;
812 try {
813 m_construct(new_start + elems_before, std::forward<Args>(args)...);
814 new_finish = pointer();
815 new_finish = m_uninitialized_copy(old_start, pos.base(), new_start);
816 ++new_finish;
817 new_finish = m_uninitialized_copy(pos.base(), old_finish, new_finish);
818 } catch (...) {
819 if (new_finish == nullptr) {
820 m_destroy(new_start, new_start + elems_before);
821 } else {
822 m_destroy(new_start, new_finish);
823 }
824 m_deallocate(new_start, len);
825 throw;
826 }
827 m_destroy(old_start, old_finish);
828 m_deallocate(old_start, m_end_of_storage - old_start);
829 m_start = new_start;
830 m_finish = new_finish;
831 m_end_of_storage = new_start + len;
832 }

888 void m_default_append(size_type n) {
889 if (n != 0) {
890 const size_type sz = size();
891 size_type navail = m_end_of_storage - m_finish;
892 if (navail >= n) {
893 m_finish = tstl::_uninitialized_default_construct_n_a(m_finish, n, m_alloc);
894 } else {
895 const size_type len = m_check_len(n);
896 pointer new_start = m_allocate(len);
897 pointer destroy_from = nullptr;
898 try {
899 tstl::_uninitialized_default_construct_n_a(new_start + sz, n, m_alloc);
900 destroy_from = new_start + sz;

902 } catch (...) {
